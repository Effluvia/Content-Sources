GUI_ACTION
	NAME ExitMovie
	# $param0 is 0 for escaped, 1 for finished
	
	GUI_COMMAND EQUALS exitedMovie 0 GUI_ACTION_MOVIE_VARIABLE themovie FREE END
	GUI_COMMAND EQUALS exitedMovie 0 GUI_ACTION_VCR_VARIABLE movieTimer PAUSE END
	GUI_COMMAND EQUALS exitedMovie 0 LaunchMainMenu $param0 END
	
	GUI_COMMAND GUI_ACTION_SET_VARIABLE exitedMovie 1 END
	GUI_COMMAND GUI_ACTION_EXIT END 
END

GUI_ACTION
	NAME TryExitMovie
	# $param0 is 0 for escaped, 1 for finished
	
	GUI_COMMAND EQUALS intro_can_exit 1 ExitMovie $param0 END
END

GUI_ACTION
	NAME _PlayMovie
	# $param0 is the movie number to play
	
	GUI_COMMAND GUI_ACTION_MOVIE_VARIABLE themovie STOP END
	GUI_COMMAND EQUALS $param0 0 GUI_ACTION_SET_VARIABLE themovie "movies/EA_logo.avi" END
	GUI_COMMAND EQUALS $param0 1 GUI_ACTION_SET_VARIABLE themovie "movies/MGM_logo.avi" END
#	GUI_COMMAND EQUALS $param0 2 GUI_ACTION_SET_VARIABLE themovie "movies/GBX_logo.avi" END

	GUI_COMMAND GUI_ACTION_SET_VARIABLE_VARIABLE currentMovieNum $param0 END
	
	GUI_COMMAND GUI_ACTION_VCR_VARIABLE movieTimer REWIND END
	GUI_COMMAND GUI_ACTION_VCR_VARIABLE movieTimer PLAY END
	GUI_COMMAND GUI_ACTION_MOVIE_VARIABLE themovie PLAY END
END

GUI_ACTION
	NAME PlayMovie
	
	GUI_COMMAND EQUALS exitedMovie 0 _PlayMovie $param0 END
END

GUI_ACTION
	NAME PlayNextMovie
	
	GUI_COMMAND GUI_ACTION_ADD_VARIABLE currentMovieNum 1 END
	GUI_COMMAND GREATERTHAN currentMovieNum 1 ExitMovie 1 END
	GUI_COMMAND EQUALS currentMovieNum 2 PlayMovie currentMovieNum END
	GUI_COMMAND LESSTHAN currentMovieNum 2 PlayMovie currentMovieNum END
END

GUI_ACTION
	NAME InitializeIntro
	
	GUI_COMMAND GUI_ACTION_DIALOG LOADACTION GUI_SETTINGS_FILE END
	GUI_COMMAND GUI_SETTINGS_FILE LOAD "nightfire"" END
	GUI_COMMAND GUI_SETTINGS_FILE VARIABLE intro_can_exit END
	GUI_COMMAND GUI_SETTINGS_FILE CLOSE END
	
	GUI_COMMAND GUI_ACTION_VCR_VARIABLE movieTimer REWIND END
	GUI_COMMAND GUI_ACTION_VCR_VARIABLE movieTimer PLAY END
	
	GUI_COMMAND GUI_ACTION_SET_VARIABLE exitedMovie 0 END
	GUI_COMMAND GUI_ACTION_SET_VARIABLE initialized 1 END
	GUI_COMMAND GUI_ACTION_SET_VARIABLE currentMovieNum 0 END
	GUI_COMMAND PlayMovie currentMovieNum END
END

GUI_ACTION
	NAME TryPlayNextMovie
	
	GUI_COMMAND EQUALS initialized 1 PlayNextMovie END
	GUI_COMMAND EQUALS initialized 0 InitializeIntro END
END
