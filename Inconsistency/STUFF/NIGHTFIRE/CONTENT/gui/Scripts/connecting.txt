GUI_ACTION
	NAME GetHotAction
	# $param0 is the help text
	# $param1 set to non-zero to play the hotsound
	GUI_COMMAND SetHelpText $param0 END
	GUI_COMMAND GREATERTHAN $param1 0 PlayHotSound END
END

	NAME PlaySelect

END

GUI_ACTION
	NAME ChangeButtonFocus
	# $param0 == string name of button to change focus to

	GUI_COMMAND PlaySelect END
	GUI_COMMAND GUI_ACTION_SET_FOCUS $param0 END
END

	NAME STD_BUTTON_ACTION
	GUI_COMMAND EQUALS $param0 1 ChangeButtonFocus THIS_GENERIC END
	GUI_COMMAND NOTEQUALS $param0 1 GUI_ACTION_SET_FOCUS THIS_GENERIC END

GUI_ACTION
	NAME AskForPassword
	
	GUI_COMMAND GUI_ACTION_DIALOG OPENMENU "gui/Scripts/Dialogs/dlg_password.txt" END
	GUI_COMMAND GUI_ACTION_DIALOG SETVARIABLE "gui/Scripts/Dialogs/dlg_password.txt" from_connecting_screen 1 END
END

GUI_ACTION
	NAME CheckToStartTimeout
	
	GUI_COMMAND GUI_ACTION_SET_VARIABLE TEMP_VAR 0 END
	GUI_COMMAND GREATERTHAN con_status 3 GUI_ACTION_SET_VARIABLE TEMP_VAR 1 END
	GUI_COMMAND EQUALS con_host_error 1 GUI_ACTION_SET_VARIABLE TEMP_VAR 1 END
	
	GUI_COMMAND EQUALS TEMP_VAR 1 GUI_ACTION_SET_VARIABLE con_attempt_valid 0 END
	GUI_COMMAND EQUALS TEMP_VAR 1 GUI_ACTION_VCR_VARIABLE con_timer REWIND END
	GUI_COMMAND EQUALS TEMP_VAR 1 GUI_ACTION_VCR_VARIABLE con_timer PLAY END
END
GUI_ACTION
	NAME CheckTimeout
	
	GUI_COMMAND GREATERTHAN con_timer 5 AbortConnection END
	GUI_COMMAND GREATERTHAN con_timer 5 GUI_ACTION_SET_VARIABLE con_timer PAUSE END
	GUI_COMMAND GREATERTHAN con_timer 5 GUI_ACTION_VCR_VARIABLE con_timer REWIND END
END

GUI_ACTION
	NAME StatusChanged
	
	GUI_COMMAND EQUALS con_status 3 AskForPassword END
	GUI_COMMAND GUI_ACTION_SET_VARIABLE_VARIABLE old_con_status con_status END
END
GUI_ACTION
	NAME FrameUpdate
	
	GUI_COMMAND EQUALS isInitialized 0 ConnectingInitialize END
	
	GUI_COMMAND NOTEQUALS con_status old_con_status StatusChanged END
	GUI_COMMAND EQUALS con_attempt_valid 1 CheckToStartTimeout END
	GUI_COMMAND EQUALS con_attempt_valid 0 CheckTimeout END
END
GUI_ACTION
	NAME AbortConnection
	
	GUI_COMMAND GUI_ACTION_EXECUTE ADDLINE "CL_Disconnect" END
	GUI_COMMAND GUI_ACTION_DIALOG OPENMENU _gui_mainmenu END
	GUI_COMMAND GUI_ACTION_EXIT END
END
