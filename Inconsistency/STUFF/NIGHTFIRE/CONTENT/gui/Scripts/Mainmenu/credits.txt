GUI_ACTION
	NAME Credits_Init
	
	GUI_COMMAND GUI_ACTION_DIALOG LOADACTION GUI_ACTION_MUSIC END
	GUI_COMMAND GUI_ACTION_DIALOG LOADACTION GUI_FILE_PARSER END
	GUI_COMMAND GUI_FILE_PARSER OPEN "gui/Scripts/Mainmenu/creditslist.txt" END	
	GUI_COMMAND GUI_ACTION_MUSIC PLAY gui ambient frontend_ectest.ogg 0 100 5000 5000 END

	GUI_COMMAND GUI_ACTION_SET_VARIABLE credits_initialized 1 END
	GUI_COMMAND GUI_ACTION_VCR_VARIABLE credits_timer REWIND END
	GUI_COMMAND GUI_ACTION_VCR_VARIABLE credits_timer PLAY END
	GUI_COMMAND GUI_ACTION_SET_VARIABLE old_credits_timer -1 END
	GUI_COMMAND GUI_FILE_PARSER REWIND END
	GUI_COMMAND Credits_LoadNewData END
END
GUI_ACTION
	NAME Credits_Stop
	
	GUI_COMMAND EQUALS credits_initialized 1 GUI_FILE_PARSER CLOSE END
	GUI_COMMAND EQUALS credits_initialized 1 GUI_ACTION_VCR_VARIABLE credits_timer REWIND END
	GUI_COMMAND EQUALS credits_initialized 1 GUI_ACTION_VCR_VARIABLE credits_timer PAUSE END
	GUI_COMMAND EQUALS credits_initialized 1 GUI_ACTION_SET_VARIABLE credits_initialized 0 END
	GUI_COMMAND EQUALS credits_initialized 1 GUI_ACTION_SET_VARIABLE old_credits_timer -1 END
	GUI_COMMAND GUI_ACTION_EXIT END
END
