GUI_ACTION
	NAME EditBoxInputAccept
	# $param0 is the string variable
	# $param1 is the control
	# $param2 is an action that sets up the editing parameters

	GUI_COMMAND GUI_ACTION_TEXT_ENTRY_HELPER ISEDITING $param0 TEMP_VAR END
	GUI_COMMAND EQUALS TEMP_VAR 0 BeginTextEntry $param0 $param1 $param2 END
	GUI_COMMAND EQUALS TEMP_VAR 1 GUI_ACTION_TEXT_ENTRY_HELPER ENDENTRY 1 END
END
GUI_ACTION
	NAME BeginTextEntry
	# $param0 is the string variable
	# $param1 is the control
	# $param2 is an action that sets up the editing parameters
	
	# first stop our current selection
	GUI_COMMAND GUI_ACTION_TEXT_ENTRY_HELPER ISEDITING ANY TEMP_VAR END
	GUI_COMMAND EQUALS TEMP_VAR 1 GUI_ACTION_TEXT_ENTRY_HELPER ENDENTRY 1 END
	
	# start editing the new one
	GUI_COMMAND $param2 END
	GUI_COMMAND GUI_ACTION_TEXT_ENTRY_HELPER BEGINENTRY $param0 END
	GUI_COMMAND GUI_ACTION_SET_FOCUS $param1 END
END
GUI_ACTION
	NAME EndTextEntry

	GUI_COMMAND GUI_ACTION_TEXT_ENTRY_HELPER ISEDITING $param0 TEMP_VAR END
	GUI_COMMAND EQUALS TEMP_VAR 1 GUI_ACTION_TEXT_ENTRY_HELPER ENDENTRY $param1 END
END

GUI_VARIABLE_FLOAT
	NAME TEMP_VAR_2
	INITIAL_VALUE 0
	MIN_VALUE -99999
	MAX_VALUE 99999
END
GUI_ACTION
	NAME ValidateFileIndex
	# $param0 is the file index
	# $param1 is the validated result
	# $param2 will hold the real index computed
	
	# get the real index and store it away
	GUI_COMMAND GetFileListIndex $param0 TEMP_VAR_2 END
	GUI_COMMAND GUI_ACTION_SAVE_LOAD_HELPER GETFIELD EXISTS TEMP_VAR TEMP_VAR_2 END
	
	GUI_COMMAND EQUALS TEMP_VAR_2 0 GUI_ACTION_SET_VARIABLE TEMP_VAR 1 END		# always OK the 'new' selection

	GUI_COMMAND EQUALS TEMP_VAR 0 GUI_ACTION_SET_VARIABLE_VARIABLE selIndex lastselIndex END	
	GUI_COMMAND EQUALS TEMP_VAR 1 GUI_ACTION_SET_VARIABLE selIndex $param0 END
	GUI_COMMAND EQUALS TEMP_VAR 1 GUI_ACTION_SET_VARIABLE_VARIABLE $param2 TEMP_VAR_2 END
END
GUI_ACTION
	NAME SelectFileIndex
	# $param0 is the index to try to select
	
	GUI_COMMAND NOTEQUALS $param0 -1 ValidateFileIndex $param0 selIndex realIndex END
	GUI_COMMAND EQUALS $param0 -1 GUI_ACTION_SET_VARIABLE selIndex -1 END
	GUI_COMMAND GUI_ACTION_SET_VARIABLE_VARIABLE lastselIndex selIndex END
	
	GUI_COMMAND EQUALS selIndex -1 GUI_ACTION_SET_VARIABLE realIndex -1 END
	GUI_COMMAND EQUALS selIndex -1 GUI_ACTION_SET_VARIABLE selValid 0 END
	GUI_COMMAND NOTEQUALS selIndex -1 GUI_ACTION_SET_VARIABLE selValid 1 END
END
GUI_ACTION
	NAME ClickSelectFileIndex
